clc;
clear all;
close all;
am=5;
fp=4;
fc=50;
ac=10;
t=0:0.001:1;
x=am.*sin(2*pi*fc*t);
subplot(3,1,1);
plot(t,x);
xlabel('time');
ylabel('amplitude');
title('Amplitude/AA/21');
y=am/2.*square(2*pi*fp*t)+(am/2);
subplot(3,1,2);
plot(t,y);
xlabel('time');
ylabel('amplitude');
title('message signal/AA/21');

z=x.*y;
subplot(3,1,3);
plot(t,z);
xlabel('time');
ylabel('amplitude');
title('ASK/AA/21');
